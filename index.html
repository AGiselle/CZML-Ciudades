// Crear visor Cesium const viewer = new Cesium.Viewer("cesiumContainer", { imageryProvider: new Cesium.OpenStreetMapImageryProvider({ url: "https://a.tile.openstreetmap.org/" }), baseLayerPicker: false, timeline: true, animation: true }); // Cargar CZML const czmlDataSource = new Cesium.CzmlDataSource(); viewer.dataSources.add(czmlDataSource); // Ruta al archivo CZML (asegúrate que está en la misma carpeta) czmlDataSource.load("ciudades.czml").then(() => { console.log("CZML cargado correctamente"); viewer.zoomTo(czmlDataSource); }).catch((error) => { console.error("Error al cargar CZML:", error); }); // Botón Play document.getElementById("playBtn").addEventListener("click", () => { viewer.clock.shouldAnimate = true; });
